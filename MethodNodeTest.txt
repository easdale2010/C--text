public class MethodNodeTest {
	public static void main(String[] args) {
		String name="Main";
		MethodNode node=new MethodNode(name);
		int option;
		boolean found;
		System.out.println(node);
		do{
			System.out.println("0: quit");
			System.out.println("1: add method to current");
			System.out.println("2: Get method in current");
			System.out.println("3: display current");
			System.out.println("4: display methods");
			System.out.println("5: search for method to select");
			option=Input.getInteger("input option: ");
			switch(option){
				case 1:	name=Input.getString("input new method name: ");
				node.addSubordinate(name);
						break;
				case 2:	name=Input.getString("input the method name: ");
						found=node.findWorker(name);
						if(!found)
							System.out.println("not found");
						else
							System.out.println(node.getCurrent());
						break;
				case 3:	System.out.println(node.getCurrent());
						break;
				case 4:	System.out.println(node);
						break;
				case 5: name=Input.getString("input the method name: ");
						found=node.findWorker(name);
						if(!found)
							System.out.println("not found");
						else
							System.out.println(node.selectCalledMethod(name));
						break;
			}
		}while(option!=0);
	}

}
